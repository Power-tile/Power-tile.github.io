const int R_1 = 3;
const int G_1 = 5;
const int B_1 = 6;
const int R_2 = 9;
const int G_2 = 10;
const int B_2 = 11;

void led1(int r, int g, int b) {
    analogWrite(R_1, r);
    analogWrite(G_1, g);
    analogWrite(B_1, b);
}

void led2(int r, int g, int b) {
    analogWrite(R_2, r);
    analogWrite(G_2, g);
    analogWrite(B_2, b);
}

void wait(float t) {
    delay(trunc(t*1000));
}

void setup() {
    pinMode(R_1, OUTPUT);
    pinMode(G_1, OUTPUT);
    pinMode(B_1, OUTPUT);
    pinMode(R_2, OUTPUT);
    pinMode(G_2, OUTPUT);
    pinMode(B_2, OUTPUT);
}
