const int R = 9;
const int G = 10;
const int B = 11;

const int SOUNDPIN = A0;

void led(int r, int g, int b) {
    analogWrite(R, r);
    analogWrite(G, g);
    analogWrite(B, b);
}

void setup() {
    Serial.begin(9600);
    
    pinMode(R, OUTPUT);
    pinMode(G, OUTPUT);
    pinMode(B, OUTPUT);

    pinMode(SOUNDPIN, INPUT);
}

void loop() {
    int lambda = digitalRead(SOUNDPIN);
    if (lambda) {
        led(255, 0, 0);
    } else {
        led(0, 255, 0);
    }
    Serial.println(lambda);
}
